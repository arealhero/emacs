** Helper functions
#+begin_src emacs-lisp
  (defun run-fzf ()
    (interactive)
    (let ((process-environment
	   (cons (concat "FZF_DEFAULT_COMMAND=rg --files --ignore -nH --no-heading")
		 process-environment)))
      (fzf/start default-directory #'fzf/action-find-file)))
#+end_src
