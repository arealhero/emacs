** Backup & autosave files

By default Emacs creates backup files (e.g. ~test.c~~) and autosave files (e.g. ~#test.c#~) in the same directory
the file belongs to. The next code block moves those files to the separate directory.

#+begin_src emacs-lisp
  (defconst emacs-cache-dir (expand-file-name "~/.cache/emacs"))
  (make-directory emacs-cache-dir t)

  (setq backup-directory-alist
	`((".*" . ,emacs-cache-dir)))
  (setq auto-save-file-name-transforms
	`((".*" ,emacs-cache-dir t)))
  (setq auto-save-list-file-prefix
	emacs-cache-dir)

  (setq undo-tree-history-directory-alist
	`((".*" . ,emacs-cache-dir)))
#+end_src

** Helper functions
#+begin_src emacs-lisp
  (defun run-fzf ()
    (interactive)
    (let ((process-environment
	   (cons (concat "FZF_DEFAULT_COMMAND=rgf")
		 process-environment)))
      (fzf/start default-directory #'fzf/action-find-file)))

  (defun display-current-time ()
    (interactive)
    (message (format-time-string "%Y-%m-%d %H:%M:%S")))
#+end_src
